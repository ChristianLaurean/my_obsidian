```bash
docker --version # version de docker instalada
docker #Documentación
docker info #informacion del hardware
docker images # puedo ver las imagenes que tengo
docker ps # son todos los contenedores que tengo corriendo en este momento.
docker ps -a son contenedores que no se están usando y los que si.
docker images --help #ayuda con algo
docker build -t nombre_imagen donde_se_escuentra_dockerfile
docker run 
# Eliminar contenedores

docker rm ID_contenedor.
docker rmi -f ID # eliminar de forma forzoza
```
